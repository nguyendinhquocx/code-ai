def format_large_numbers(value):
    """Format large numbers with appropriate units"""
    if value >= 1e9:
        return f'{value/1e9:.1f}B'
    elif value >= 1e6:
        return f'{value/1e6:.1f}M'
    elif value >= 1e3:
        return f'{value/1e3:.1f}K'
    return f'{value:.0f}'
